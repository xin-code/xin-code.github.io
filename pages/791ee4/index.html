常用配置项

<template>
  <div ref="chart" style="width: 800px; height: 350px"></div>
</template>

<script>
export default {
  data() {
    return {
      chartData: [
        { name: "Mon", value: 0 },
        { name: "Tue", value: 200 },
        { name: "Wed", value: 150 },
        { name: "Thu", value: 80 },
        { name: "Fri", value: 70 },
        { name: "Sat", value: 110 },
        { name: "Sunday", value: 130 },
      ],
    };
  },
  mounted() {
    this.initChart();
  },
  methods: {
    initChart() {
      const chart = echarts.init(this.$refs.chart);
      const option = {
        tooltip: {
          show: true,
          trigger: "axis",
          confine: true,
          axisPointer: {
            type: "shadow",
            shadowStyle: {
              color: "rgba(86, 255, 218, 0.11)",
            },
          },
          formatter: function (params) {
            // 悬浮文字信息
            let str = `${params[0].name}`;
            // tooltip 过滤为 0 的数据
            params.map((item) => {
              if (item.data !== 0) {
                str += `<br>${item.marker}<span style="margin-right:10px">${item.seriesName}</span><span style="float:right;font-weight:bold">${item.data}</span>`;
              }
            });
            return str;
          },
        },
        grid: {
          top: "20px",
          left: "40x",
          right: 0,
          bottom: "40px",
        },
        // 图例过多时展示翻页
        legend: {
          x: "right", // 水平居右
          y: "center", // 垂直居中
          icon: "rect", // 图例icon为方块
          backgroundColor: "transparent",
          itemHeight: 10, // 图例icon高度
          itemWidth: 19, // 图例icon宽度
          height: "90%",
          borderColor: "rgba(91, 122, 216, 0.5)",
          borderWidth: 1,
          top: 4,
          borderRadius: 4,
          itemGap: 13,
          padding: [5, 10, 10, 10],
          orient: "vertical", // 垂直排列
          type: "scroll", // 可滚动翻页的图例
          pageIconSize: 10, //翻页按钮大小
          pageIconColor: "#2C86FF", //翻页箭头颜色
          pageIconInactiveColor: "rgba(44,132,251,0.40)", //翻页（即翻页到头时箭头的颜色
          pageTextStyle: {
            color: "#999", //翻页数字颜色
          }, //开启tooltip
          tooltip: {
            show: true,
          },
          // 文字大于4个字时，切割字符，加...
          formatter: function (value) {
            if (value.length > 4) {
              return `${value.slice(0, 4)}...`;
            }
            return value;
          },
          textStyle: {
            lineHeight: 13,
          },
          align: "left", // 图例icon在左侧
          selector: [
            {
              type: "all",
              title: "全选",
            },
            {
              type: "inverse",
              title: "反选",
            },
          ],
          selected: {},
        },
        xAxis: {
          type: "category",
          data: this.chartData.map((x) => x.name),
          axisTick: {
            alignWithLabel: true,
          },
          axisLabel: {
            interval: 0, // 强制显示所有标签
            rotate: 40,
            // x轴文字过长设置超出隐藏显示省略号(超过4个字显示省略号)
            formatter: function (value) {
              if (value.length > 4) {
                return `${value.slice(0, 4)}...`;
              }
              return value;
            },
          },
        },
        yAxis: {
          type: "value",
        },
        series: [
          {
            name: "测试",
            data: this.chartData.map((x) => x.value),
            type: "bar",
            itemStyle: {
              normal: {
                label: {
                  show: true, //开启显示
                  position: "top", //在上方显示
                  // 柱状图顶端不展示数值为 0 的
                  formatter: function (val) {
                    if (val.value == 0) {
                      return "";
                    }
                  },
                  textStyle: {
                    //数值样式
                    color: "black",
                    fontSize: 12,
                  },
                },
              },
            },
          },
          {
            name: "测试名称123",
            data: this.chartData.map((x) => x.value),
            type: "bar",
            itemStyle: {
              normal: {
                label: {
                  show: true, //开启显示
                  position: "top", //在上方显示
                  // 柱状图顶端不展示数值为 0 的
                  formatter: function (val) {
                    if (val.value == 0) {
                      return "";
                    }
                  },
                  textStyle: {
                    //数值样式
                    color: "black",
                    fontSize: 12,
                  },
                },
              },
            },
          },
        ],
      };
      chart.setOption(option);
    },
  },
};
</script>
